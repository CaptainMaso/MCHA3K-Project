 83function [Theta, dTheta, Bias, P] = AccCorrection(Theta_p, dTheta_p, Bias_p, P_p, yt, rt)
 84    fm = get_fimath();
 85
 86    global T;
 87    Pbb_p = fi(P_p(3,3), 0, 16, 13, fm);
 88    Ptb_p = fi(P_p(2,3), 1, 16, 9, fm);
 89    Ptt_p = fi(P_p(2,2), 0, 16, 6, fm);
 90    Pwb_p = fi(P_p(1,3), 1, 16, 12, fm);
 91    Pwt_p = fi(P_p(1,2), 0, 16, 10, fm);
 92    Pww_p = fi(P_p(1,1), 0, 16, 13, fm);    
 93
 94    %% Kalman Gain Update
 95    kw = fi(fi_div(Pwt_p, (Ptt_p + fi_div(rt, T))), 0, 16, 29, fm);
 96    kt = fi(fi_div(Ptt_p, (Ptt_p + fi_div(rt, T))), 0, 16, 26, fm);
 97    kb = fi(fi_div(Ptb_p, (Ptt_p + fi_div(rt, T))), 1, 16, 28, fm);
 98    
 99    %% State Update
100    dTheta = fi(dTheta_p + kw*(yt-Theta_p), 1, 16, 4, fm);
101    Theta =  fi(Theta_p  + kt*(yt-Theta_p), 1, 16, 4, fm);
102    Bias =   fi(Bias_p   + kb*(yt-Theta_p), 1, 16, 9, fm);
103    
104    %% Covariance Update
105    Pww = fi(Pww_p + (fi(-2, 1, 8, 0, fm)*kw)*Pwt_p + kw^2*Ptt_p + fi_div((kw^2*rt), T), 0, 16, 13, fm);
106    Pwt = fi((fi(1, 0, 8, 0, fm) - kt)*Pwt_p + (kw*(fi_signed(kt) - fi(1, 0, 8, 0, fm)))*Ptt_p + fi_div((kt*kw*rt), T), 0, 16, 10, fm);
107    Pwb = fi((fi_uminus(kb))*Pww_p + (kb*kw)*Pwt_p + (fi(1, 0, 8, 0, fm) - kb)*Pwb_p + (kw*(kb - fi(1, 0, 8, 0, fm)))*Ptb_p + fi_div((kb*kw*rt), T), 1, 16, 12, fm);
108    Ptt = fi((fi_signed(kt) - fi(1, 0, 8, 0, fm))^2*Ptt_p + fi_div((kt^2*rt), T), 0, 16, 6, fm);
109    Ptb = fi((kb*(fi_signed(kt) - fi(1, 0, 8, 0, fm)))*Pwt_p + ((kb - fi(1, 0, 8, 0, fm))*(fi_signed(kt) - fi(1, 0, 8, 0, fm)))*Ptb_p + fi_div((kb*kt*rt), T), 1, 16, 9, fm);
110    Pbb = fi(kb^2*Pww_p + (fi(2, 0, 8, 0, fm)*kb*(kb - fi(1, 0, 8, 0, fm)))*Pwb_p + (kb - fi(1, 0, 8, 0, fm))^2*Pbb_p + fi_div((kb^2*rt), T), 0, 16, 13, fm);
111    
112    P = fi([fi(Pww, 1, 16, 5, fm), Pwt, Pwb; fi(Pwt, 1, 16, 5, fm), Ptt, Ptb; fi(Pwb, 1, 16, 9, fm), Ptb, Pbb], 1, 16, 5, fm);
113end
