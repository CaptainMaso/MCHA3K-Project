147function y = fi_signed(a)
148    coder.inline( 'always' );
149    if isfi( a ) && ~(issigned( a ))
150        nt = numerictype( a );
151        new_nt = numerictype( 1, nt.WordLength + 1, nt.FractionLength );
152        y = fi( a, new_nt, fimath( a ) );
153    else
154        y = a;
155    end
156end
