134function c = fi_div(a,b)
135    coder.inline( 'always' );
136    if isfi( a ) && isfi( b ) && isscalar( b )
137        a1 = fi( a, 'RoundMode', 'fix' );
138        b1 = fi( b, 'RoundMode', 'fix' );
139        c1 = divide( divideType( a1, b1 ), a1, b1 );
140        c = fi( c1, numerictype( c1 ), fimath( a ) );
141    else
142        c = a/b;
143    end
144end
